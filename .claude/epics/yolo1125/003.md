# Task: 建立前端頁面與鏡頭存取功能

## Description

建立前端 HTML 介面，實作 WebRTC 鏡頭存取、Canvas 繪製即時畫面，以及 WebSocket 通訊模組。這是使用者互動的主要介面，需提供清晰的視覺佈局和流暢的鏡頭畫面。

**核心目標**：
- 建立三欄式響應式佈局（使用者資料、鏡頭畫面、購物車）
- 實作 WebRTC 存取電腦鏡頭
- Canvas 顯示即時影像並支援繪製偵測框
- WebSocket 客戶端連線和訊息處理
- 基本 UI 樣式和視覺效果

## Acceptance Criteria

- [ ] 開啟 `http://localhost:8000` 顯示完整介面
- [ ] 瀏覽器請求鏡頭權限並成功存取
- [ ] Canvas 即時顯示鏡頭畫面（流暢無卡頓）
- [ ] WebSocket 成功連線至後端
- [ ] 每 200ms 擷取一影格並透過 WebSocket 傳送
- [ ] 三欄式佈局清晰顯示（左側使用者、中央鏡頭、右側購物車）
- [ ] 響應式設計，適應不同螢幕尺寸

### JavaScript - 鏡頭模組（camera.js）

```javascript
class CameraManager {
    constructor(videoId, canvasId) {
        this.video = document.getElementById(videoId);
        this.canvas = document.getElementById(canvasId);
        this.ctx = this.canvas.getContext('2d');
        this.stream = null;
        this.isRunning = false;
    }

    async start() {
        try {
            // 請求鏡頭權限
            this.stream = await navigator.mediaDevices.getUserMedia({
                video: {
                    width: { ideal: 640 },
                    height: { ideal: 480 },
                    facingMode: 'user'
                },
                audio: false
            });

            this.video.srcObject = this.stream;
            this.isRunning = true;

            // 等待視訊載入
            await new Promise((resolve) => {
                this.video.onloadedmetadata = () => {
                    this.canvas.width = this.video.videoWidth;
                    this.canvas.height = this.video.videoHeight;
                    resolve();
                };
            });

            console.log('✅ 鏡頭啟動成功');
            this.hideError();
            this.startDrawing();

        } catch (error) {
            console.error('❌ 鏡頭存取失敗:', error);
            this.showError();
        }
    }

    startDrawing() {
        const draw = () => {
            if (!this.isRunning) return;

            // 繪製當前影格到 Canvas
            this.ctx.drawImage(
                this.video,
                0, 0,
                this.canvas.width,
                this.canvas.height
            );

            requestAnimationFrame(draw);
        };
        draw();
    }

    captureFrame() {
        if (!this.isRunning) return null;

        // 擷取當前 Canvas 內容為 Base64
        return this.canvas.toDataURL('image/jpeg', 0.8);
    }

    drawBox(x, y, width, height, label, color = '#00ff00') {
        // 繪製偵測框（之後會用到）
        this.ctx.strokeStyle = color;
        this.ctx.lineWidth = 3;
        this.ctx.strokeRect(x, y, width, height);

        // 繪製標籤
        this.ctx.fillStyle = color;
        this.ctx.fillRect(x, y - 25, width, 25);
        this.ctx.fillStyle = '#000';
        this.ctx.font = '16px Arial';
        this.ctx.fillText(label, x + 5, y - 7);
    }

    showError() {
        document.getElementById('camera-error').style.display = 'block';
    }

    hideError() {
        document.getElementById('camera-error').style.display = 'none';
    }

    stop() {
        this.isRunning = false;
        if (this.stream) {
            this.stream.getTracks().forEach(track => track.stop());
        }
    }
}

// 全域實例
const cameraManager = new CameraManager('video', 'canvas');
```

### JavaScript - WebSocket 模組（websocket.js）

```javascript
class WebSocketClient {
    constructor(url) {
        this.url = url;
        this.ws = null;
        this.sessionId = this.generateSessionId();
        this.connected = false;
        this.messageHandlers = {};
    }

    generateSessionId() {
        return 'session_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);
    }

    connect() {
        const wsUrl = `${this.url}/${this.sessionId}`;
        this.ws = new WebSocket(wsUrl);

        this.ws.onopen = () => {
            console.log('✅ WebSocket 連線成功');
            this.connected = true;
            this.updateStatus(true);
        };

        this.ws.onclose = () => {
            console.log('❌ WebSocket 斷線');
            this.connected = false;
            this.updateStatus(false);
            // 自動重連
            setTimeout(() => this.connect(), 3000);
        };

        this.ws.onerror = (error) => {
            console.error('❌ WebSocket 錯誤:', error);
        };

        this.ws.onmessage = (event) => {
            const data = JSON.parse(event.data);
            this.handleMessage(data);
        };
    }

    send(data) {
        if (this.connected && this.ws.readyState === WebSocket.OPEN) {
            this.ws.send(JSON.stringify(data));
        }
    }

    sendFrame(frameData) {
        this.send({
            type: 'frame',
            frame: frameData,
            timestamp: new Date().toISOString()
        });
    }

    on(messageType, handler) {
        this.messageHandlers[messageType] = handler;
    }

    handleMessage(data) {
        const handler = this.messageHandlers[data.type];
        if (handler) {
            handler(data);
        }
    }

    updateStatus(connected) {
        const statusEl = document.getElementById('connection-status');
        if (connected) {
            statusEl.textContent = '● 已連線';
            statusEl.className = 'status-connected';
        } else {
            statusEl.textContent = '● 未連線';
            statusEl.className = 'status-disconnected';
        }
    }
}

// 全域實例
const wsClient = new WebSocketClient('ws://localhost:8000/ws');
```

### JavaScript - 主程式（main.js）

```javascript
// 初始化
document.addEventListener('DOMContentLoaded', async () => {
    console.log('🚀 啟動前端系統');

    // 啟動鏡頭
    await cameraManager.start();

    // 連線 WebSocket
    wsClient.connect();

    // 定期發送影格
    setInterval(() => {
        if (wsClient.connected) {
            const frame = cameraManager.captureFrame();
            if (frame) {
                wsClient.sendFrame(frame);
            }
        }
    }, 200); // 每 200ms 發送一影格（5 FPS）

    // 監聽訊息
    wsClient.on('face_detected', (data) => {
        console.log('👤 偵測到人臉:', data.user);
        updateUserInfo(data.user);
    });

    wsClient.on('product_detected', (data) => {
        console.log('📦 偵測到商品:', data.product);
        cart.addItem(data.product);
    });

    wsClient.on('cart_updated', (data) => {
        console.log('🛒 購物車更新');
        cart.update(data.cart);
    });

    // 結帳按鈕
    document.getElementById('checkout-btn').addEventListener('click', () => {
        cart.checkout();
    });
});

function updateUserInfo(user) {
    const userInfoEl = document.getElementById('user-info');
    userInfoEl.innerHTML = `
        <div class="user-details">
            <h3>歡迎，${user.name}</h3>
            <p>📞 ${user.phone}</p>
        </div>
    `;
}
```
